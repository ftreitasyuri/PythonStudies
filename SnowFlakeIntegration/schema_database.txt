-- 1) Warehouse leve e com auto pause
CREATE WAREHOUSE IF NOT EXISTS WH_INGEST
  WAREHOUSE_SIZE = XSMALL
  AUTO_SUSPEND = 60
  AUTO_RESUME = TRUE
  INITIALLY_SUSPENDED = TRUE;

-- 2) Database e Schema
CREATE DATABASE IF NOT EXISTS DATALAKE;
CREATE SCHEMA IF NOT EXISTS DATALAKE.RAW;

-- 3) Tabela destino (ajuste tipos se quiser)
CREATE TABLE IF NOT EXISTS DATALAKE.RAW.CHECKLIST (
  ARQUIVO STRING,
  IMPRESSAO_GERADA TIMESTAMP_NTZ,
  TITULO STRING
);

-- 4) Usar o que criamos
USE WAREHOUSE WH_INGEST;
USE DATABASE DATALAKE;
USE SCHEMA RAW;
